query NodeQuery(
  $id: ID!
) {
  node(id: $id) {
    id
    ... on Story {
      comments(first: 10) @__clientField(handle: "storyComments") {
        edges {
          node {
            actor {
              name
              friends(first: 10) @__clientField(handle: "authorFriends") {
                edges {
                  node {
                    name
                  }
                }
                ... on FriendsConnection {
                  edges {
                    cursor
                    node {
                      __typename
                    }
                  }
                  pageInfo {
                    endCursor
                    hasNextPage
                    hasPreviousPage
                    startCursor
                  }
                }
              }
            }
          }
        }
        ... on CommentsConnection {
          edges {
            cursor
            node {
              __typename
            }
          }
          pageInfo {
            endCursor
            hasNextPage
            hasPreviousPage
            startCursor
          }
        }
      }
    }
  }
}
